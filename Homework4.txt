# Вводим верхнюю границу числа 
n=int(input("Введите целое число"))
# Создаем картеж в диапазоне от 1 до n
E=tuple(range(1, n+1))
# Создаем картеж из нечетных чисел картежа Е
A = tuple(s for s in E if s%2!=0)
# Создаем множество
B = set()
# Создаем цикал для создания множества в котором будут попарно нечетные числа {(1,3), (3,5), (5,7) и т.д.}
for k in range(len(A)-1):
    C = A[k], A [k+1]
    B.add(C)
print(B)

# Создаем верхнюю границу списка
n = int(input("Введите целое число"))
# Создаем список A в диапазоне от 1 до n
A = list(range(1, n+1))
# Создаем списог В в обратной последовательности от списка А
B = list(reversed(A))
# Создаем пустой словар
S = dict()
# Создаем цикал для создания словаря из ключей из списка А и значения из списка В
for k in range(len(A)):
    # Создаем словарь из одного ключа и одного значения
    D=dict(zip([A[k]],[B[k]]))
    # Добовляем все полученые словари в один словарь
    S.update(D)
print(S)
